\begin{hwexercise}
Given a paracompact space $X$, along the natural bijections
\begin{align*}
    \Prin_{O(k)}(X) \leftrightarrow \check{H}^1(X,O(k)) \leftrightarrow \Vect_\R^k(X),
\end{align*}
show that a real rank $r$ vector bundle $E \to X$ is associated to its frame bundle $\Fr(E)$.
\end{hwexercise}

\begin{hwexercise}
Verify that the \emph{tautological} bundle on the infinite Grassmannian is associated to the infinite Stiefel manifold.
\end{hwexercise}
